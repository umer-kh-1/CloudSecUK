Driver utility
                
                    pnputil.exe

                    pnputil /enum-drivers # lists all non-MS packages
                    
                    pnputil add-driver d:\usbcam\USBCAM.inf  # adds a package that USBCAM.inf specifies.
                    
                    pnputil add-driver c:\drivers*.inf # adds all packages in c:\drivers
                    
                    pnputil /delete-driver oem0.inf /uninstall
WMIC
                
                    wmic

                    wmic memphysical get maxcapacity

                    wmic:root\cli>/output:c:\system.txt product get name,version

                    wmic path softwareLicensingService get OA3xOriginalProductKey

                    wmic bios get serialnumber

                    wmic > product get name > product where name =“insert name of application” call uninstall
